No se conecta a internet
# Las interfaces son las conexiones a internet que tenemos, pueden ser por wifi o por ethernet

# Para listar las interfaces ls /sys/class/net # Siempre existe una interface 'lo', el resto son nuestras conexiones

# Editar la configuracion de las interfaces sudo nano /etc/network/interfaces # Comentar las lineas que digan 'iface NOMBRECONEXION inet...'

# Para comentar una linea, simplemente poner '#' al inicio de esta
