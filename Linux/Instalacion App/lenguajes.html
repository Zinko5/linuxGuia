Instalar lenguajes de programacion
C# (.NET 6.0.4xx)
# Opcion 1: sudo apt install dotnet6
# Opcion 2: wget https://packages.microsoft.com/config/ubuntu/22.04/packages-microsoft-prod.deb -O packages-microsoft-prod.deb
sudo dpkg -i packages-microsoft-prod.deb
rm packages-microsoft-prod.deb # Crear un archivo para priorizar los repositorios PMC touch /etc/apt/preferences
cd /etc/apt nano preferences # pegar con Ctrl + Shift + V Package: *
Pin: origin "packages.microsoft.com"
Pin-Priority: 1001
 # Presionar Ctrl + X, S, Enter

# Instalar el SDK 6.0 de .NET sudo apt update
sudo apt install dotnet-sdk-6.0
Python 3
sudo apt install python3-pip
