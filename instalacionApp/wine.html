Wine
# Instalar wine sudo apt install -y wine64 wine32 # Instalar algunas dependencias utiles sudo apt install -y libasound2-plugins:i386 libsdl2-2.0-0:i386 libdbus-1-3:i386 libsqlite3-0:i386 # Tambien es recomendable instalar wine-tricks sudo apt install winetricks # Existe manera mas manual y *oficial* de instalar wine, pero puede presentar errores, ver Guia oficial
