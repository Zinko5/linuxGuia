Instalar lenguajes de programacion
C# (.NET 6.0.4xx)
wget https://packages.microsoft.com/config/ubuntu/22.04/packages-microsoft-prod.deb -O packages-microsoft-prod.deb
sudo dpkg -i packages-microsoft-prod.deb
rm packages-microsoft-prod.deb # Crear un archivo para priorizar los repositorios PMC sudo touch /etc/apt/preferences
cd /etc/apt
sudo nano preferences # pegar con Ctrl + Shift + V Package: *
Pin: origin "packages.microsoft.com"
Pin-Priority: 1001
 # Presionar Ctrl + X, S, Enter

# Instalar el SDK 6.0 de .NET sudo apt update
sudo apt install dotnet-sdk-6.0
Python 3
sudo apt install python3-pip
Java 17 (recomendado)
sudo apt install openjdk-17-jdk
Java 11 (antiguo pero puede ser mas estable para algunos proyectos)
sudo apt install default-jdk
