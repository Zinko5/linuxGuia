Cambiar username
# Cambiar el username y user's groupname, no se puede hacer logeado en el mismo usuario al que se le cambiara el username sudo usermod -l NuevoUsername ViejoUsername
sudo groupmod -n NuevoUsername ViejoUsername # Renombrar al directorio de home sudo usermod -d /home/NuevoUsername -m NuevoUsername
